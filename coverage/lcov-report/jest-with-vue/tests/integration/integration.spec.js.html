import { render, cleanup } from '@testing-library/vue'
import storeConfig from '@/store/store-config'
import Todos from '@/components/Todos.vue'
 
// automatically unmount and cleanup DOM after the test is finished.
afterEach(cleanup)
 
function renderVuexTestComponent(customStore) {
  // Render the component and merge the original store and the custom one
  // provided as a parameter. This way, we can alter some behaviors of the
  // initial implementation.
  return render(Todos, { store: { ...storeConfig, ...customStore } })
}
 
describe('integration tests', () => {
  it('Renders todo list', () => {
    const { getByText } = renderVuexTestComponent()
    expect(getByText('Async Add'))
  })
  it('Async add button renders', () => {
    const { getByText } = renderVuexTestComponent({
      state: { todos: [{ id: 1, title: 'test1', completed: false }] }
    })
    expect(getByText('Async Add'))
  })
})
